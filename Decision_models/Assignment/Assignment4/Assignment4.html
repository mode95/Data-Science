Introduzione

Il task assegnato consiste nell’implementare un o più algoritmi per trovare una soluzione sub-ottimale al Flow shop scheduling problem . In questo caso è stato scelto il Genetic Algorithm e il Tabu Search . In seguito il risultato ottenuto sarà confrontato con il migliore conosiuto (presente sul sito Taillard ) e con quello ottenuto con un algoritmo già presente in \(R\) (il Genetic Algorithm della libreria \(GA\)).

Il problema

Nel Flow shop scheduling problem abbiamo \(n\) macchine e \(m\) lavori da compiere. Ogni lavoro deve essere processato su tutte le macchine con lo steso ordine. In oltre due macchine non possono compiere lo stesso lavoro simultaneamente e il tempo per ogni lavoro cambia da macchina a macchina. L’obiettivo è quello di minimizzare il \(makespan\) che corrisponde al tempo totale di esecuzione.

Makespan

Si implementa la funzione che data la matrice di lavori e macchine (rappresentante il tempo imiegato da ogni macchina per effettuare i lavori) calcola il \(makespan\).

set.seed(4062019)
jobs_time <- function(mat, seq){
  mat <- mat[, seq]
  m <- matrix(0, nrow = nrow(mat), ncol = ncol(mat))
  
  for (row in 1:nrow(mat)) {
    for (col in 1:ncol(mat)) {
      # prima cella
      if(row == 1 & col == 1){
        m[row, col] <- mat[row, col]
      }
      
      # prima riga
      else if(row == 1){
        m[row, col] <- mat[row, col] + m[row, col - 1]
      }
      
      else{
        if(col == 1){
          m[row, col] <- m[row-1, col] + mat[row, col]
        }
        else{
          if(m[row, col-1] <= m[row-1, col]){
            m[row, col] <- mat[row, col] +  m[row-1, col]
          }
          else{
            m[row, col] <- mat[row, col] +  m[row, col-1]
          }
        }
      }
    }
  }
  return(1/as.integer(m[nrow(m), ncol(m)]))
}

L’algoritmo genetico

L’algoritmo genetico è basato concettualmente sui principi che regolano l’evoluzione naturale delle specie. L’idea di base quinidi è quella avviare ricorsivamente l’algoritmo e di selezionare le soluzioni migliori ad ogni iterazione e di ricombinarle in qualche modo (\(crossover\) e \(mutazione\)) in maniera che evolvano verso un punto di ottimo. La funzione da massimizzare si chiama \(fit\) \(funtion\) (nel caso corrente è \(\frac {1}{makespan}\) ) e l’insieme degli elementi su cui si valuta ad ogni iterazione è chiamato \(popolazione\).

generateRandomWork <- function(){
  vet <- c(1:nWorks)                                         # crea una permutazione casuale
  randperm(vet) 
}

generateRandomPopulation <- function(popSize){               # crea una popolazione  casuale
  t(replicate(popSize, generateRandomWork()))
}

evaluateWork <- function(fitFun, works){                     # valuta il fit factor di un individuo
  fitFun(matrice, works)
}

evaluatePopulation <- function(pop){
  data.frame(index = c(1:nrow(pop)),
             fit = apply(pop, 1,
                         curry(evaluateWork, jobs_time)))    # valuta il fit factor della popolazione
}

populationStats <- function(gen, scores){
  bestScore <- max(scores$fit)
  avgScore <- mean(scores$fit)
  medianScore <- median(scores$fit)
  item = list(gen = gen, best = bestScore, avg = avgScore, median =medianScore )
  class(item)<- "traceItem"
  return(item)
}
rouletteSelection <- function(scores, pop){
  idx = sample_n(scores, size = 2, replace = F, weight = scores$fit)$index
  pop[idx, ]
}

Si implementano ora anche i metodi per la \(mutazione\):

swapMutation <- function(p){
  pos = sample(nWorks, 2)
  p[c(pos[1], pos[2])] = p[c(pos[2], pos[1])]
  p
}

E il \(crossover\):

generateOffspring <- function(p1, p2){
  swath = round(nWorks/2)
  cut1 = sample(nWorks - swath, 1)
  cut2 = cut1 + swath-1
  off1 = rep(0, nWorks)
  sel = p1[cut1:cut2]
  off1[cut1:cut2]= sel
  r = p2[!p2 %in% sel] 
  r1= r[c(1:cut1-1)]
  r2= r[c(cut1:length(r))]
  off1[c(1:cut1-1)] = r1
  off1[c((cut2+1):length(off1))] = r2
  off1
}

L’algoritmo vero e proprio è:

simpleGA <- function(popSize = 200, crossProb = 0.8, mutProbability = 0.2, numGenerations = 1000) {
  pop <- generateRandomPopulation(popSize)
  trace = NULL
  for (gen in c(1:numGenerations)) {
    newPop = NULL
    scores <- evaluatePopulation(pop)

    newPop <- rbind(newPop, pop[head(scores[order(scores$fit,decreasing = T), ],2)$index,])
    if(!(gen %% 50) ) {
      trace <- append(trace, populationStats(gen,scores))
    }
    for (i in c(2:(popSize/2))) {
      # selection
      parents <- rouletteSelection(scores, pop)
      # crossover
      if (runif(1)<=crossProb) {
        off1 <- generateOffspring(parents[1,], parents[2,])
        off2 <- generateOffspring(parents[2,], parents[1,])
      }
      else{
        off1 <- parents[1,]; off2<- parents[2,]
      }
      newPop <- rbind(newPop, as.array(off1), as.array(off2))
    }
    # mutatazione
    for (i in c(1:popSize)) {
      if(runif(1) <= mutProbability){
        newPop[i,] <- swapMutation(newPop[i,])
      }
    }
    pop = newPop
  }
  
  pop <- as.data.frame(pop)
  pop <- pop %>%
    unite(sec, c(colnames(pop)), sep = ", ", remove = FALSE)
  pop$res <- scores$fit
  pop <- pop[, c(1, ncol(pop))]
  max <- max(pop$res)
  pop <- pop[pop$res == max, ]    # solo i valori più alti di fit function
  
  
  m <- as.numeric(1/pop$res[1])
  pop$makespan <- m
  rownames(pop) <- 1:nrow(pop)
  return(list(trace2Dataframe(trace), pop))
}

Tabu Search

Il \(Tabu\) \(Search\) è una procedura metaeuristica che migliora l’efficienza di un classico algoritmo di ricerca locale memorizzando informazioni riguardo il processo di ricerca. Ad ogni iterazione si considera l’intorno della soluzione trovata e si eliminano alcune mosse in base alla memoria dell’algoritmo. Ciò permette di sfugire alla trappola dei minimi/massimi locali. Ciò però non comporta l’assoluta certezza di trovare la soluzione ottima.

swap_Tabu <- function(v, i , j){
  
    aux <- v[i]
    v[i] <- v[j]
    v[j] <- aux
    return(v)
}
TS <- function(mat, iter, tabu.size=7, report.evol=FALSE){
  
  sol <- generateRandomWork()
    n <- length(sol)
  obj <- 1/jobs_time(mat, sol)
  evol <- numeric(iter)
  
  solbest <- sol
  objbest <- obj
  
  k <- 1
  
  tabu.list <- matrix(numeric(tabu.size*2), tabu.size, 2)
  
  
  while(k<=iter){
    
    #bisogna trovare la soluzione migliore per l'iterazione
    move <- numeric(2)
    objiter <- Inf
    soliter <- numeric(n)
    update.tabu <- FALSE

    for(i in 1:(n-1)){
      for(j in (i+1):n){
        soltest <- swap_Tabu(sol, i, j)
        objtest <- 1/jobs_time(mat, soltest)
        
        is.tabu <- FALSE
        for(s in 1:tabu.size) if(tabu.list[s,1]==i & tabu.list[s,2]==j) is.tabu <- TRUE
        
        if(is.tabu){
          #mla mossa è tabu
          
          if(objtest < objbest){
            soliter <- soltest
            objiter <- objtest
            update.tabu <- FALSE
          }
        }
        else{
          #la mossa non è tabu
          if(objtest < objiter){
            soliter <- soltest
            objiter <- objtest
            update.tabu <- TRUE
            move <- c(i,j)
          }
        }
      }
    }
    
    sol <- soliter
    obj <- objiter
    
    #update della soluzione migliore
    if(obj < objbest){
      objbest <- obj
      solbest <- sol
    }
    
    if(update.tabu){
      for(s in tabu.size:2) tabu.list[s, ] <- tabu.list[s-1, ]
      tabu.list[1, ] <- move
    }
    
    evol[k] <- obj
    k <- k+1
    
  }
  
  if(report.evol)
    return(list(sol=solbest, obj=objbest, evol=evol))
  else
    return(list(sol=solbest, obj=objbest))
}
